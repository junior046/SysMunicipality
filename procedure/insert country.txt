CREATE OR REPLACE PROCEDURE insertcountry (
    countryname IN country.country_name%TYPE
) IS
BEGIN
    INSERT INTO country (
        id_country,
        country_name
    ) VALUES (
        s_country.NEXTVAL,
        countryname
    );

    COMMIT;
EXCEPTION
    WHEN OTHERS THEN
        dbms_output.put_line('CODIGO:' || sqlcode);
        dbms_output.put_line('MENSAJE:' || sqlerrm);
END insertcountry;
/
