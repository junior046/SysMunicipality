create or replace PROCEDURE updatedistrictname (
    vciddistrict   IN   district.id_district%TYPE,
    vcname       IN   district.district_name%TYPE
) IS

    CURSOR curdistrict IS
    SELECT
        *
    FROM
        district
    FOR UPDATE;

    rowdistrict   district%rowtype;
    pdistrictid   district.id_district%TYPE;
BEGIN
    OPEN curdistrict;
    LOOP
        FETCH curdistrict INTO rowdistrict;
        EXIT WHEN curdistrict%notfound;
        IF rowdistrict.id_district = vciddistrict THEN
            UPDATE district
            SET
                district_name = vcname
            WHERE
                CURRENT OF curdistrict;

        END IF;

    END LOOP;

    CLOSE curdistrict;
    commit;
EXCEPTION
    WHEN no_data_found THEN
        dbms_output.put_line('NO EXISTE EL DISTRITO');
    WHEN OTHERS THEN
        dbms_output.put_line('CODIGO:' || sqlcode);
        dbms_output.put_line('MENSAJE:' || sqlerrm);
END updatedistrictname;
